Ejemplos prácticos de manejo de cadenas

Ejemplo 1 – Python (Verificar palíndromo)

def es_palindromo(texto):
    texto = texto.lower().replace(" ", "").replace(",", "").replace(".", "")
    return texto == texto[::-1]

print(es_palindromo("Anita lava la tina"))  # True
print(es_palindromo("Hola mundo"))          # False

Ejemplo 2 – Java (Búsqueda de subcadena con indexOf)

public class BuscarCadena {
    public static void main(String[] args) {
        String texto = "El algoritmo KMP es eficiente";
        String patron = "KMP";
        
        int posicion = texto.indexOf(patron);
        if (posicion != -1) {
            System.out.println("Encontrado en la posición: " + posicion);
        } else {
            System.out.println("No encontrado");
        }
    }
}
